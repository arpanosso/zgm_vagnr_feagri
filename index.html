# criando fun√ß√£o para estat√≠stica descritiva
estat_desc <- function(x){
  x<-na.omit(x)
  media <- mean(x)
  mediana <- median(x)
  dp <- sd(x)
  cv <- 100*dp/media
  MIN <- min(x)
  MAX <- max(x)
  q1 <- quantile(x,0.25)
  q3 <- quantile(x,0.75)
  assi <- agricolae::skewness(x)
  curt <- agricolae::kurtosis(x)
  c(M√≠nimo = MIN, Q1=q1, M√©dia=media, Med=mediana, Q3=q3, M√°ximo=MAX,
    DesvPad = dp, CV = cv,Assimetria=assi,Curtose=curt)
}
